83  void set_shift(const Scalar& sigmar, const Scalar& sigmai)
84  {
85  // Create a sparse idendity matrix (1 + 0i on diagonal)
86  SparseComplexMatrix I(m_n, m_n);
87  I.setIdentity();
88  // Sparse LU decomposition
89  m_solver.compute(m_mat.template cast<Complex>() - Complex(sigmar, sigmai) * I);
90  // Set cache to zero
91  m_x_cache.resize(m_n);
92  m_x_cache.setZero();
93  }
94 
102  // y_out = Re( inv(A - sigma * I) * x_in )
103  void perform_op(const Scalar* x_in, Scalar* y_out) const
104  {
105  m_x_cache.real() = MapConstVec(x_in, m_n);
106  MapVec y(y_out, m_n);
107  y.noalias() = m_solver.solve(m_x_cache).real();
108  }
